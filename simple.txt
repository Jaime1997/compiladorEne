program simple;

func int sub(int x, int y){
    int result;
    result = x - y;
    return result;
}

func int add(int x, int y){
    int result;
    result = x + y;
    return result;
}

main(){
    int a, b, c, d, e;

    a = 5;
    b = 4;

    if(a>b){
        print("true");
    } else {
        print("false");
    }

    c = 2*(a+b)+2;

    d = add(a,b);
    e = sub(c,d);

    print(a,b,c,d,e);
}
